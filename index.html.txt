/* ==========================================================================
   🚨 ปรับฟังก์ชันยิงข้อมูลขึ้นเซฟออนไลน์บน Google Sheets
   ========================================================================== */
function doCIDirect(s, status){
  var overlay = document.getElementById('loading-overlay');
  if(overlay) overlay.style.display = 'flex'; // แสดงแถบกำลังบันทึกกูเกิลชีต

  var now = new Date();
  var t = String(now.getHours()).padStart(2,'0') + ':' + String(now.getMinutes()).padStart(2,'0');
  var workDateIso = now.getFullYear() + '-' + String(now.getMonth()+1).padStart(2,'0') + '-' + String(now.getDate()).padStart(2,'0');

  var payload = {
    uid: s.uid,
    name: s.fn + ' ' + s.ln,
    br: s.br,
    dp: s.dp,
    ps: s.ps,
    et: s.et,
    status: status,
    calendarTime: now.toLocaleString('th-TH'),
    workDate: workDateIso
  };

  // ดำเนินการยิงข้อมูลผ่าน Fetch API ไปยังหลังบ้านออนไลน์
  fetch(SCRIPT_URL, {
    method: "POST",
    mode: "no-cors", // ป้องกันเรื่องปัญหา CORS policy ข้ามโดเมนบนเบราว์เซอร์
    cache: "no-cache",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(payload)
  }).then(function() {
    if(overlay) overlay.style.display = 'none';
    
    // บันทึกใส่ประวัติหน้ากากเครื่องเพื่อโชว์ Real-time
    logs.unshift({name:s.fn+' '+s.ln, br:s.br, dp:s.dp, status:status, time:t, workDate:workDateIso});
    renderLogs();
    
    document.getElementById('r-icon').textContent = status==='Check-in'?'✅':'👋';
    document.getElementById('r-status').textContent = status;
    document.getElementById('r-status').style.color = status==='Check-in'?'#029664':'#DC2626';
    document.getElementById('r-name').textContent = s.fn+' '+s.ln;
    document.getElementById('r-info').textContent = s.br+' · '+s.dp;
    document.getElementById('r-time').textContent = '⏰ '+t+' (ผูกกะ: '+workDateIso+')';
    
    openM('m-result');
    setTimeout(resetS, 200);
    
    // โหลดรีเฟรชยอดรวม Dashboard อัตโนมัติหลังบันทึก
    setTimeout(loadLiveLogs, 1000); 
  }).catch(function(err) {
    if(overlay) overlay.style.display = 'none';
    alert('บันทึกผิดพลาดภายนอก กรุณาเช็คการเชื่อมต่อเน็ตคลังสินค้าครับ');
  });
}

/* ==========================================================================
   🚨 ดึงประวัติ Logs แบบเรียลไทม์สดๆ จาก Google Sheets มาโชว์ตัวเลข Dashboard
   ========================================================================== */
function loadLiveLogs() {
  if(SCRIPT_URL === "วางลิงก์เว็บแอปที่ได้จากกูเกิลตรงนี้") return;
  
  fetch(SCRIPT_URL)
    .then(function(res) { return res.json(); })
    .then(function(data) {
       // จัดระเบียบแปลงเวลาที่ดึงมาจากชีตให้อยู่ในฟอร์แมตแดชบอร์ด
       logs = data.map(function(item) {
         return {
           name: item.name, br: item.br, dp: item.dp, status: item.status,
           time: item.time ? item.time.split(' ')[1] || '00:00' : '00:00',
           workDate: item.workDate
         };
       });
       renderDash();
       renderLogs();
    }).catch(function(e){ console.log("กำลังรอรับส่งยอดกะครั้งแรก..."); });
}

function renderDash(){
  var branches=actBr.length?actBr:BR;
  var all=logs; // ใช้ยอดจริงออนไลน์ทั้งหมด
  var tIn=0,tOut=0,tTotal=0;
  var grid=document.getElementById('d-grid');grid.innerHTML='';
  
  branches.forEach(function(b){
    var bl=all.filter(function(l){return l.br===b});
    var bIn=bl.filter(function(l){return l.status==='Check-in'}).length;
    var bOut=bl.filter(function(l){return l.status==='Check-out'}).length;
    tIn+=bIn;tOut+=bOut;tTotal+=bIn+bOut;
    
    var dh='';
    DEPTS.forEach(function(dp){
      var dl=bl.filter(function(l){return l.dp===dp});if(!dl.length)return;
      var dI=dl.filter(function(l){return l.status==='Check-in'}).length;
      var dO=dl.filter(function(l){return l.status==='Check-out'}).length;
      dh+='<div class="dept-row"><span style="color:'+DC[dp]+';font-weight:600;font-size:11px">'+dp+'</span><span style="font-size:11px"><span style="color:#029664;font-weight:600">'+dI+' in</span><span style="color:#CBD5E1;margin:0 3px">/</span><span style="color:#E88A05;font-weight:600">'+dO+' out</span></span></div>';
    });
    
    var brName=b.replace(/^ST\d+\s/,'');var brCode=b.split(' ')[0];
    var card=document.createElement('div');card.className='br-card';
    card.onclick=function(){var dl=card.querySelector('.dept-list');dl.classList.toggle('show');card.classList.toggle('open');};
    card.innerHTML=
      '<div class="br-name">'+brName+'</div>'+
      '<div class="br-code">'+brCode+'</div>'+
      '<div class="br-nums">'+
      '<div class="br-num"><div class="br-num-val" style="color:#029664">'+bIn+'</div><div class="br-num-lbl">Check-in</div></div>'+
      '<div class="br-num"><div class="br-num-val" style="color:#E88A05">'+bOut+'</div><div class="br-num-lbl">Check-out</div></div>'+
      '</div>'+
      '<div class="br-attend" style="font-size:10px;color:#64748B">รวม '+(bIn+bOut)+' คน • อัตราเข้างาน '+Math.round((bIn/(bIn+bOut||1))*100)+'%</div>'+
      '<div style="font-size:10px;color:#94A3B8;text-align:center;margin-bottom:2px"><i class="ti ti-chevron-down" style="font-size:11px"></i> กดดูรายแผนก</div>'+
      '<div class="dept-list">'+dh+'</div>';
    grid.appendChild(card);
  });
  
  document.getElementById('d-in').textContent=tIn;
  document.getElementById('d-out').textContent=tOut;
  document.getElementById('d-br').textContent=branches.length;
  document.getElementById('d-total').textContent=tTotal;
}

function showQR(uid){
  var s=staff.find(function(x){return x.uid===uid});if(!s)return;
  document.getElementById('qr-name').textContent=s.fn+' '+s.ln+' • '+s.uid;
  var cv=document.getElementById('qr-cv');var ctx=cv.getContext('2d');
  ctx.fillStyle='#fff';ctx.fillRect(0,0,140,140);ctx.fillStyle='#000';
  var cell=5;
  [[0,0],[14,0],[0,14]].forEach(function(p){ctx.fillRect(p[0]*cell+5,p[1]*cell+5,7*cell,7*cell);ctx.fillStyle='#fff';ctx.fillRect((p[0]+1)*cell+5,(p[1]+1)*cell+5,5*cell,5*cell);ctx.fillStyle='#000';ctx.fillRect((p[0]+2)*cell+5,(p[1]+2)*cell+5,3*cell,3*cell);ctx.fillStyle='#000';});
  var h=0,d=uid+'|'+s.fn+'|'+s.br;for(var i=0;i<d.length;i++)h=(h*31+d.charCodeAt(i))&0xFFFFFF;
  for(var r=2;r<19;r++)for(var c=2;c<19;c++){if((r<9&&c<9)||(r<9&&c>12)||(r>12&&c<9))continue;if((h>>(r+c))&1)ctx.fillRect(c*cell+5,r*cell+5,cell,cell);}
  openM('m-qr');
}

function renderStaff(){

  var q=(document.getElementById('sl-q')||{}).value||'';

  var filtered=staff.filter(function(s){ return(!q||(s.fn+' '+s.ln).toLowerCase().indexOf(q.toLowerCase())>=0)&&(!br||s.br===br); });
  
  document.getElementById('r-cnt').textContent=staff.length;
  document.getElementById('r-filtered-cnt').textContent=filtered.length;
  
  if(!filtered.length){el.innerHTML='<div style="text-align:center;padding:14px;color:#94A3B8;font-size:13px">ไม่มีพนักงานในกลุ่มนี้</div>';return;}
  el.innerHTML=filtered.map(function(s){

      '<div class="avatar" style="width:38px;height:38px;font-size:15px;background:linear-gradient(135deg,#667EEA,#764BA2);color:#fff">'+s.fn[0].toUpperCase()+'</div>'+

      '</div></div></div>';
  }).join('');
}
